void Delay(unsigned int xms)		//@11.0592MHz
{
	unsigned char i, j;
    	while(xms){
		_nop_();
		i = 2;
		j = 199;
		do
		{
		while (--j);
		 } while (--i);
		xms--;
	}
}
  
void Delay1us(unsigned int mms)		//@11.0592MHz
{
	while(mms){
	_nop_();
	_nop_();
	_nop_();
	mms--;
  }		
}

//----------串口中断初始化函数-----------------     
 
void usart_init()
 
{
    SCON = 0x50;                  //REN=1允许串行接受状态，串口工作模式1                      
    TMOD = 0x20;                  //定时器工作方式2                    
    PCON = 0x00;                                                          
    TH1  = 0xFD;                  //波特率9600，数据为8，停止位1 (11.0592M)
    TL1  = 0xFD;                                                           
    ES   = 1;                     //开串口中断                  
    EA   = 1;                     //开总中断         
    TR1  = 1;                     //启动定时器
}
 
//---------------------------
 
//---------串口数据发送-------------------
 
void send_data(unsigned char a){ 
 
    SBUF = a;                      //SUBF接受/发送缓冲器
    while(0 == TI);                //等待发送完毕，再执行
    TI=0;                         //手动清0
}
 
void send_data_cc( int a){ 
 
  char bai,shi,ge,fen;
  int shu;
	shu=a;   	

//	bai=shu/1000;
//	shi=shu/100%10;
//	ge=shu/10%10;
//	fen=shu%10;
	
	bai = chartonumber(shu/1000);
	shi = chartonumber(shu/100%10);
	ge = chartonumber(shu/10%10);
	fen = chartonumber(shu%10);

	SBUF = bai;//SUBF接受/发送缓冲器
	while(!TI); //等待发送完毕，再执行
	TI=0;      //手动清0
	SBUF = shi;
	while(!TI);
	TI=0;
	SBUF = ge;
	while(!TI);
	TI=0;
	SBUF = fen;
	while(!TI);
	TI=0;	
				                        
}
