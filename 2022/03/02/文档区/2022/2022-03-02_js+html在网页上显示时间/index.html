js文件代码

window.onload=function()
{
	window.requestAnimationFrame(getDate)
}
 
function getDate()
{
	window.setTimeout(function(){
		window.requestAnimationFrame(getDate)
	},1000/2)
	var time=new Date();   
	var year=time.getFullYear()  //年
	var month=time.getMonth()+1;  //月
	var day=time.getDay()    //日
	var days=time.getDate() //日期
	var hour=time.getHours()   //小时
	var minute=time.getMinutes()  //分钟
	var second=time.getSeconds()   //秒
	if(month<10) month="0"+month
	if(days<10) days="0"+days
	if(hour<10) hour="0"+hour
	if(minute<10) minute="0"+minute
	if(second<10) second="0"+second
	var weekshow=new Array("日","一","二","三","四","五","六")
	var HT=document.getElementById("time")   //设置html文件,访问ID
	var timeshow1="日期:  "+year+"-"+month+"-"+days+"  "+"星期"+weekshow[day]
	var timeshow2="时间:"+hour+" : "+minute+" :"+second 
	HT.innerHTML= timeshow1//显示日期
	//HT.innerHTML= timeshow2//显示时间,自行更改
 
}
